עשרת דיברות פולישוק

ויהי בימים ההם

ויראו בני האדם כי העתיד נסתר מעיניהם, ויבקשו לדעת את אשר לא נודע, ואת אשר טרם בא לעולם.

ויקומו להם שווקים, ויקומו להם מנבאים, ויקומו להם מומחים מטעם עצמם.

וירבו הדעות בארץ.

ויאמרו זקני פולישוק:

אלה הדברים אשר יעשה האדם וחי בהם, ואלה הדברים אשר יעשה ויפסיד בהם.

הדיבר הראשון

אָנֹכִי פּוֹלִישׁוּק אֲשֶׁר נְתַתִּיךָ גִּישָׁה אֶל הַפְּלַטְפוֹרְמָה.

לֹא תְצַפֶּה מִמֶּנִּי לָשֵׂאת בְּאַחֲרָיוּת אֲשֶׁר לָךְ.

כִּי אַתָּה תִּשָּׂא בְּאַחֲרָיוּת לְכָל מַעֲשֶׂיךָ.

וְלִתְחוּשַׁת הַבִּטָּחוֹן שֶׁהָיְתָה לְךָ לִפְנֵי שֶׁנִּתְבָּרֵר כִּי טָעִיתָ.

הדיבר השני

לֹא תַעֲשֶׂה לְךָ אֱלֹהִים מִכָּל גְּרָף וּמִכָּל תַּרְשִׁים.

לֹא תִשְׁתַּחֲוֶה לָהֶם וְלֹא תַעָבְדֵם.

כִּי גְּרָף יָכוֹל לְהֵרָאוֹת חָכָם וּבְכָל זֹאת לִהְיוֹת שָׁגוּי.

וְיֵשׁ אֲשֶׁר הִפְסִידוּ מָמוֹן רַב בְּשֵׁם הַצִּיר הָעוֹלֶה.

הדיבר השלישי

לֹא תִשָּׂא אֶת שֵׁם הַמּוּמְחֶה לַשָּׁוְא.

כִּי רַבִּים הֵם הַמַּכְרִיזִים עַל עַצְמָם מוּמְחִים.

וּמְעַטִּים הֵם הַיּוֹדְעִים מָה יֵלֵד יוֹם.

וְאִם כָּתַב אִישׁ כִּי "מֵאָה אָחוּז" —

הִזָּהֵר מִמֶּנּוּ כִּפְלַיִם.

הדיבר הרביעי

זָכוֹר אֶת שִׁקּוּל הַדַּעַת לְקַדְּשׁוֹ.

שֵׁשׁ פְּעָמִים תִּקְרָא.

שֵׁשׁ פְּעָמִים תִּבְדּוֹק.

וּבַפַּעַם הַשְּׁבִיעִית תָּבִין שֶׁאַתָּה עֲדַיִן לֹא יוֹדֵעַ.

הדיבר החמישי

כַּבֵּד אֶת אַחֲרָיוּתְךָ וְאֶת תּוֹצְאוֹתֶיהָ.

לְמַעַן יַאֲרִיכוּ יָמֶיךָ בְּוִכּוּחִים בְּקוֹבוּצוֹת הַוָּאטְסְאַפּ.

כִּי הַמַּעֲשֶׂה מַעֲשֶׂיךָ.

וְהַתּוֹצָאָה תּוֹצָאָתְךָ.

וְהָאַחֲרָיוּת אַחֲרָיוּתְךָ.

הדיבר השישי

לֹא תַאֲשִׁים אֶת הַפְּלַטְפוֹרְמָה.

לֹא עַל הֶפְסֵדֶיךָ.

לֹא עַל טָעוּיוֹתֶיךָ.

לֹא עַל הַדְּבָרִים שֶׁנִּרְאוּ לְךָ רַעְיוֹן מֻצְלָח בְּשָׁעָה שְׁתַּיִם בַּלַּיְלָה.

כִּי הַפְּלַטְפוֹרְמָה נָתְנָה לְךָ כְּלִי.

וְאַתָּה בָּחַרְתָּ מָה לַעֲשׂוֹת בּוֹ.

הדיבר השביעי

לֹא תִנְאָף אַחֲרֵי מֵידָע שֶׁלֹּא נִדְרַשׁ.

וְאִם נָתַתָּ מִסְפַּר טֶלֶפוֹן.

אוֹ כְּתוֹבֶת.

אוֹ סוֹד מִשְׁפַּחְתִּי.

אוֹ סִסְמָה.

אוֹ מַפְתֵּחַ פְּרָטִי.

וְלֹא בִּקְּשׁוּ אוֹתָם מִמְּךָ —

דַּע כִּי מִיָּדְךָ נִתְּנוּ.

וְעָלֶיךָ תָּשׁוּב אַחֲרָיוּתָם.

הדיבר השמיני

לֹא תִגְנֹב אַחֲרָיוּת מֵאֲחֵרִים וְלֹא תַעֲבִירָהּ לַפְּלַטְפוֹרְמָה.

כִּי אִישׁ בְּחֶטְאוֹ יִשָּׂא.

וְאִישׁ בְּטָעוּתוֹ יִשָּׂא.

וְאִישׁ בְּהִתְלַהֲבוּתוֹ יִשָּׂא.

הדיבר התשיעי

לֹא תַעֲנֶה בַפְּלַטְפוֹרְמָה עֵדוּת כִּי "הָיִיתִי בָּטוּחַ".

כִּי בִּטָּחוֹן אֵינוֹ רְאָיָה.

וְוַדָּאוּת אֵינָהּ הוֹכָחָה.

וְיֵשׁ אֲשֶׁר הָיוּ בְּטוּחִים מְאֹד וְנִמְצְאוּ טוֹעִים מְאֹד.

הדיבר העשירי

לֹא תַחְמֹד אֶת אַחֲרָיוּתוֹ שֶׁל רֵעֶךָ.

לֹא תַחְמֹד אֶת הַמַּנְהֵל.

וְלֹא אֶת הַמַּפְתֵּחַ.

וְלֹא אֶת הַמּוֹדֶרָטוֹר.

וְלֹא אֶת הַמִּשְׁתַּמֵּשׁ אֲשֶׁר כָּתַב "סָגוּר בְּמֵאָה אָחוּז".

כִּי אֶת אַחֲרָיוּתְךָ תִּשָּׂא.

וְאֶת אַחֲרָיוּתָם יִשְּׂאוּ הֵם.

ספר הברית

וַיְהִי כִּי נִתְּנָה לָאָדָם גִּישָׁה אֶל פּוֹלִישׁוּק.

וַיֹּאמֶר:

"אֲנִי מְאַשֵּׁר כִּי קָרָאתִי."

וְלֹא קָרָא.

וַיֹּאמֶר:

"אֲנִי מֵבִין."

וְלֹא הֵבִין.

וַיֹּאמֶר:

"אֲנִי מְסַכִּים."

וְעַל כֵּן נֶחְשַׁב כְּמַסְכִּים.

וַתִּהְיֶה הָאַחֲרָיוּת עָלָיו.

וְלֹא עַל פּוֹלִישׁוּק.

וַיַּרְא פּוֹלִישׁוּק כִּי טוֹב.
